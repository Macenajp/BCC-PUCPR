CREATE DATABASE IF NOT EXISTS stormmotors;

USE stormmotors;

CREATE TABLE IF NOT EXISTS clientes (
	id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(200),
    endereco VARCHAR(200),
    email VARCHAR(200),
    cpf VARCHAR(200),
    senha VARCHAR(200),
    telefone VARCHAR(200)
);

CREATE TABLE IF NOT EXISTS carros (
	id INT AUTO_INCREMENT PRIMARY KEY,
    modelo VARCHAR(200),
    cor VARCHAR(200),
    motor VARCHAR(200),
    ano VARCHAR(200),
    cambio VARCHAR(200),
    combustivel VARCHAR(200),
    descricao VARCHAR(200),
    valor VARCHAR(200)
);

CREATE TABLE IF NOT EXISTS pedidos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    nome_carro VARCHAR(255),
    cor VARCHAR(100),
    preco VARCHAR(200),
    imagem VARCHAR(255),
    data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES clientes(id)
);

CREATE TABLE IF NOT EXISTS administradores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL
);

CREATE TABLE IF NOT EXISTS status_pedidos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT NOT NULL,
    status VARCHAR(100) NOT NULL DEFAULT 'Em Processamento', 
    data_status DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS pagamentos (
    id INT AUTO_INCREMENT PRIMARY KEY,                     
    id_pedido INT NOT NULL,                                 
    valor_pago VARCHAR(200),                     
    status_pagamento VARCHAR(50) DEFAULT 'pagamento realizado', 
    data_hora DATETIME DEFAULT CURRENT_TIMESTAMP,          
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS pre_vendas (
    id INT AUTO_INCREMENT PRIMARY KEY,                   
    id_usuario INT NOT NULL,                             
    nome_carro VARCHAR(255) NOT NULL,                    
    valor VARCHAR(200) NOT NULL,                           
    data_pre_venda DATETIME DEFAULT CURRENT_TIMESTAMP,     
    status VARCHAR(50) DEFAULT 'pré-venda registrada',     
    FOREIGN KEY (id_usuario) REFERENCES clientes(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS montadora (
    id INT AUTO_INCREMENT PRIMARY KEY,             
    id_pedido INT NOT NULL, 
    nome VARCHAR(255) NOT NULL DEFAULT 'Storm Motors - Brasil',         
    pais VARCHAR(100) DEFAULT 'Brasil',
    nome_time VARCHAR(100) NOT NULL DEFAULT 'Time A',                 
    status_montagem VARCHAR(50) DEFAULT 'Em preparação', 
    data_inicio DATETIME DEFAULT CURRENT_TIMESTAMP,      
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id) ON DELETE CASCADE
);
